var iNow = new Date();

// var iNew = new Date( 2013, 10, 27, 21,56,0 );

var iNew = new Date( 'November 27,2013 22:3:0' );

var t = Math.floor((iNew - iNow)/1000);

// 毫秒 - 秒

alert( str );

// 时：Math.floor(t%86400/3600)
